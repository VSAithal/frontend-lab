# Input

Inputs collect user text. Use `InputGroup` when you need icons or addons.

## Usage
```tsx
import { Input } from '@vsaithal/core-ui'

export function Example() {
  return <Input placeholder="you@example.com" aria-label="Email" />
}
```

## Input Group
```tsx
import {
  InputGroup,
  InputGroupAddon,
  InputGroupInput,
} from '@vsaithal/core-ui'

export function Example() {
  return (
    <InputGroup>
      <InputGroupAddon position="left">$</InputGroupAddon>
      <InputGroupInput placeholder="0.00" aria-label="Amount" />
      <InputGroupAddon position="right">USD</InputGroupAddon>
    </InputGroup>
  )
}
```
