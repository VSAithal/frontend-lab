# Radio Group

Radio groups let users pick a single option from a list. Pass an array of options and control value if needed.

## Usage
```tsx
import { RadioGroup } from '@vsaithal/core-ui'

const options = [
  { value: 'email', label: 'Email' },
  { value: 'sms', label: 'SMS' },
]

export function Example() {
  return <RadioGroup defaultValue="email" options={options} />
}
```

## Controlled Usage
```tsx
import { useState } from 'react'
import { RadioGroup } from '@vsaithal/core-ui'

const options = [
  { value: 'email', label: 'Email' },
  { value: 'sms', label: 'SMS' },
]

export function Example() {
  const [value, setValue] = useState('email')
  return (
    <RadioGroup
      value={value}
      onValueChange={setValue}
      options={options}
    />
  )
}
```
